def request_month(): #  
    response = requests.get('http://www.it.kmitl.ac.th/~teerapong/news_archive/index.html') #
    html_page = bs4.BeautifulSoup(response.content, 'lxml')
    selector = 'li > a'         #  tags   tag  tag a
    tags_month = html_page.select(selector)  # beautifulsoup  tags  selector
    list_tag = []
                           #  tag month  path  link 
    for tag in tags_month: #  tags_month  path  link   list 
        list_tag.append('http://www.it.kmitl.ac.th/~teerapong/news_archive/'+tag['href']) # list
    return list_tag         #

def category_three(web_month): #     
    category_list = []
    for web in request_month(): #  1  12 
        count = 0
        response = requests.get(str(web))     #
        html_page = bs4.BeautifulSoup(response.content, 'lxml')
        selector_ = 'tbody > tr > td'          # tag    category  tbody  td 
        tags_category = html_page.select(selector_)     # beautifulsoup  tags  selector_
        
        for tag in tags_category: # category   list
                                                #  html_page.select   1  2 tag 
                                                #  tag category  tag   tag category
            if (count % 2 == 0) and (str(tag.text[1:]) != 'N/A'): # tag  
                                                       # tag category  count % 2 == 0  category
                                                       #   category  != N/A
                category_list.append(tag.text[1:])    # category  list
            count += 1
    return category_list # category  web


def path_text(web_month):    # 
    list_path = []
    for  web in request_month():     #   1  12
        response = requests.get(str(web)) #
        html_page = bs4.BeautifulSoup(response.content, 'lxml')
        selector_text = 'tbody > tr > td > a ' # tag    
        tags_text = html_page.select(selector_text) # beautifulsoup  tags  selector_text
                                                    # path    
        for path in tags_text:        # path   link  path 
            list_path.append('http://www.it.kmitl.ac.th/~teerapong/news_archive/'+ str(path['href'])) # list
    return list_path # web


def extract_text(web_month): #
    long_text = []
    list_head = []
    for web in path_text(request_month()): # 
        text_box = ''
        response = requests.get(web)  #
        html_page = bs4.BeautifulSoup(response.content, 'lxml')
        selector_head = 'h2' # tag  2   selector_head 
        selector_text = 'p ' #  selector_text 
        tags_body = html_page.select(selector_text) # beautifulsoup  tags  selector_text
        tags_head = html_page.select(selector_head) # beautifulsoup  tags  selector_head
        for  head in tags_head: # tags_head   list
            list_head.append(head.text)
        for text in tags_body:    #     html pharagrap  1  pharagrap 
                                  #   pharagrap
            if (text.text in list_head): #   tag p  
                pass                     #  text  list  
            elif (('class='  in str(text)) == False): # text    tag   Return to article search results
                                                      # text class=  ? 
                text_box += str(text.text)            # text  claas   string text_box 
                                                      #  pharagrap   1  
        text_box = text_box.replace("\"","")          #  \   replace  ""
        long_text.append(text_box)                    #  sting  list 
    return long_text # list

def stem_tokenizer(text):
    standard_tokenizer = CountVectorizer().build_tokenizer() # CountVectorizer
    tokens = standard_tokenizer(text)  #
    stemmer = PorterStemmer()          # PorterStemmer
    stems = []
    for token in tokens:               #
        stems.append( stemmer.stem(token.lower()) ) #  stemmer  
    return stems

# Word  Dictionary 
def lemma_tokenizer(text):
    standard_tokenizer = CountVectorizer().build_tokenizer()# CountVectorizer
    tokens = standard_tokenizer(text)                      #
    lemmatizer = nltk.stem.WordNetLemmatizer()             #  WordNetLemmatizer
    lemma_tokens = []
    for token in tokens:                                   # #
        lemma_tokens.append( lemmatizer.lemmatize(token))  #  lemmatize 
    return lemma_tokens                                    #  Word  Dictionary

def lemma_tokenizer_w_pos_tag(text):
    def convert_tags(tag):                              # tag    verb  noun
        if tag == 'vbd' or tag == 'vbg' or tag == 'vbz':
            return 'v'
        else: 
            return 'n'
    
  
    standard_tokenizer = CountVectorizer().build_tokenizer() # CountVectorizer
    tokens = standard_tokenizer(text)                        #
    tokens_with_pos_tag = nltk.pos_tag(tokens)               #  pos_tag 
    lemmatizer = nltk.WordNetLemmatizer()                    # WordNetLemmatizer
    lemma_tokens = []
    for token in tokens_with_pos_tag:                        #  tag
        new_tag = convert_tags(token[1].lower())             #  convert_tags 
        lemma_tokens.append(lemmatizer.lemmatize(token[0], new_tag))
    
    return lemma_tokens
